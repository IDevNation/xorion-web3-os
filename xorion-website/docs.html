Quick Start Example

use xorion_sdk::wallet::Wallet;
use xorion_sdk::rpc::EthereumProvider;

// Create a new wallet
let wallet = Wallet::new();
let mnemonic = wallet.mnemonic();

// Connect to Ethereum
let eth = EthereumProvider::mainnet();
let balance = eth.get_balance(wallet.eth_address()).await?;

// Send a private transaction
wallet.send_private(eth, recipient, amount).await?;
